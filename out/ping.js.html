const { SlashCommandBuilder } = require('discord.js');

/**
 * Represents the ping slash command.
 * @type {import('discord.js').ApplicationCommandData}
 */
module.exports = {
  data: new SlashCommandBuilder()
    .setName('ping')
    .setDescription('Replies with Pong!'),

  /**
   * Executes the ping command.
   * @param {import('discord.js').CommandInteraction} interaction - The interaction object.
   * @returns {Promise<void>} A Promise that resolves when the interaction is replied to.
   */
  async execute(interaction) {
    /**
     * Responds with "Pong!" to the user who triggered the command.
     * @type {import('discord.js').Message}
     */
    const reply = await interaction.reply('Pong!');
    return reply;
  },
};
