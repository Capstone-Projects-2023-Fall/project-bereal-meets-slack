const mysql = require('mysql2');
require('dotenv').config();

/**
 * Creates a connection pool for local database calls during testing.
 *
 * @returns {Object} - MySQL connection pool.
 */
function createConnectionPoolLocal() {
  const dboptions = {
    connectionLimit: 15,
    host: process.env.DB_HOST,
    user: process.env.DB_USER,
    password: process.env.DB_PASS,
    database: process.env.DB_NAME,
  };
  return mysql.createPool(dboptions);
}

/**
 * Creates a connection pool for cloud database calls.
 *
 * @returns {Object} - MySQL connection pool.
 */
function createConnectionPoolCloud() {
  const dboptions = {
    connectionLimit: 15,
    socketPath: process.env.INSTANCE_UNIX_SOCKET,
    user: process.env.DB_USER,
    password: process.env.DB_PASS,
    database: process.env.DB_NAME,
    waitForConnections: true,
    connectTimeout: 2000,
  };
  return mysql.createPool(dboptions);
}

/**
 * Creates a promise-based connection pool based on the specified environment flag.
 *
 * @returns {Object} - Promise-based MySQL connection pool.
 */
function createPromiseConnectionPool() {
  var flag = process.env.PROD_FLAG;
  var promisepool;

  if (flag === "CLOUD") {
    console.log("CLOUD check");
    const pool = createConnectionPoolCloud();
    promisepool = pool.promise();
  } else if (flag === "LOCAL") {
    console.log("LOCAL check");
    const pool = createConnectionPoolLocal();
    promisepool = pool.promise();
  }

  return promisepool;
}

/**
 * The MySQL connection pool used throughout the Discord bot.
 *
 * @type {Object} - Promise-based MySQL connection pool.
 */
const pool = createPromiseConnectionPool();

module.exports = {
  pool,
};
